35 OrbitalTransform::OrbitalTransform(const Hamiltonian& HamIn):
36  index(HamIn)
37 {
38  numberOfIrreps = index.getNirreps();
39  int L = HamIn.getL();

41  _hamorig.reset(new CheMPS2::Hamiltonian(HamIn));

43  _unitary.reset(new UnitaryMatrix(index));
44 
45  //Create the memory for the orbital transformations.
46  unsigned long long maxlinsize = 0;
47  for (int irrep=0; irrep< index.getNirreps(); irrep++)
48  {
49  unsigned int linsize_irrep = index.getNORB(irrep);
50  if (linsize_irrep > maxlinsize)
51  maxlinsize = linsize_irrep;
52  }
53 
54  //Determine the blocksize for the 2-body transformation
55  auto& maxBlockSize = maxlinsize;
56  //Allocate 2-body rotation memory: One array is approx (maxBlockSize/273.0)^4 * 42 GiB --> [maxBlockSize=100 --> 750 MB]
57  auto maxBSpower4 = maxBlockSize * maxBlockSize * maxBlockSize * maxBlockSize; //Note that 273**4 overfloats the 32 bit integer!!!
58  auto sizeWorkmem1 = max( max( maxBSpower4 , 3*maxlinsize*maxlinsize ) , 1uLL * L * L ); //For (2-body tfo , updateUnitary, calcNOON)
59  auto sizeWorkmem2 = max( max( maxBSpower4 , 2*maxlinsize*maxlinsize ) , L*(L + 1uLL) ); //For (2-body tfo, updateUnitary and rotate_to_active_space, rotate2DMand1DM)
60  mem1.reset(new double[sizeWorkmem1]);
61  mem2.reset(new double[sizeWorkmem2]);
62 }

138 void OrbitalTransform::buildOneBodyMatrixElements()
139 {
140  if(!QmatrixWork.size() || !OneBodyMatrixElements.size())
141  {
142  QmatrixWork.resize(numberOfIrreps);
143  OneBodyMatrixElements.resize(numberOfIrreps);
144 
145  for (int irrep=0; irrep<numberOfIrreps; irrep++)
146  {
147  const int size = index.getNORB(irrep) * index.getNORB(irrep);
148  QmatrixWork[irrep].reset(new double[size]);
149  OneBodyMatrixElements[irrep].reset(new double[size]);
150  }
151  }
152 
153  //#pragma omp parallel for schedule(dynamic)
154  for (int irrep=0; irrep<numberOfIrreps; irrep++)
155  {
156  const int linsize = index.getNORB(irrep);
157  for (int row=0; row<linsize; row++)
158  {
159  const int HamIndex1 = index.getNstart(irrep) + row;
160  //set the diagonal elements.
161  OneBodyMatrixElements[irrep][row*(1+linsize)] = _hamorig->getTmat(HamIndex1,HamIndex1);
162  for (int col=row+1; col<linsize; col++)
163  {
164  const int HamIndex2 = index.getNstart(irrep) + col;
165  // remember blas and lapack are in fortran and there it is convenient to run first over columnsand then over rows.
166  OneBodyMatrixElements[irrep][row + linsize * col] = _hamorig->getTmat(HamIndex1,HamIndex2);
167  OneBodyMatrixElements[irrep][col + linsize * row] = OneBodyMatrixElements[irrep][row + linsize * col];
168  }
169  }
170  }
171  rotate_old_to_new(OneBodyMatrixElements.data());

287 void OrbitalTransform::DoJacobiRotation(CheMPS2::Hamiltonian &ham_rot, int k, int l, double theta)
288 {
289  assert(ham_rot.getOrbitalIrrep(k) == ham_rot.getOrbitalIrrep(l));
290 
291  const int irrep = ham_rot.getOrbitalIrrep(k);
292  const int linsize = index.getNORB(irrep);
293  const int shift = index.getNstart(irrep);
294  // the relative index in the irrep
295  const int k2 = k - shift;
296  const int l2 = l - shift;
297 
298  const double cos = std::cos(theta);
299  const double sin = std::sin(theta);
300 
301  // the one particle integrals
302 
303  // first copy element that we are gonna overwrite
304  const double tmpkk = ham_rot.getTmat(k,k);
305  const double tmpll = ham_rot.getTmat(l,l);
306  const double tmpkl = ham_rot.getTmat(k,l);
307 
308  double tmp;
309  tmp = cos*cos*tmpkk+sin*sin*tmpll-2*cos*sin*tmpkl;
310  ham_rot.setTmat(k,k,tmp);
311 
312  tmp = cos*cos*tmpll+sin*sin*tmpkk+2*cos*sin*tmpkl;
313  ham_rot.setTmat(l,l,tmp);
314 
315  tmp = tmpkl*(cos*cos-sin*sin)+cos*sin*(tmpkk-tmpll);
316  ham_rot.setTmat(k,l,tmp);
317 
318  for(int a=shift;a<linsize+shift;a++)
319  {
320  if(a == k || a == l)
321  continue;
322 
323  const double tmpk = ham_rot.getTmat(k,a);
324  const double tmpl = ham_rot.getTmat(l,a);
325 
326  tmp = cos * tmpk - sin * tmpl;
327  ham_rot.setTmat(a,k,tmp);
328 
329  tmp = cos * tmpl + sin * tmpk;
330  ham_rot.setTmat(a,l,tmp);
331  }
332 
333 
334  // the two particle elements
335  // this is code from Sebastian (CheMPS2) adapted to
336  // Jacobi rotations in one irrep
337  // Two-body terms --> use eightfold permutation symmetry in the irreps :-)
338  for (int irrep1 = 0; irrep1<numberOfIrreps; irrep1++)
339  for (int irrep2 = irrep1; irrep2<numberOfIrreps; irrep2++)
340  {
341  const int productSymm = SymmInfo.directProd(irrep1, irrep2);
342  for (int irrep3 = irrep1; irrep3<numberOfIrreps; irrep3++)
343  {
344  const int irrep4 = SymmInfo.directProd(productSymm, irrep3);
345  // Generated all possible combinations of allowed irreps
346  if(irrep4>=irrep2 && (irrep4==irrep || irrep3==irrep || irrep2==irrep || irrep1==irrep))
347  {
348  const int linsize1 = index.getNORB(irrep1);
349  const int linsize2 = index.getNORB(irrep2);
350  const int linsize3 = index.getNORB(irrep3);
351  const int linsize4 = index.getNORB(irrep4);
352 
353  if ((linsize1>0) && (linsize2>0) && (linsize3>0) && (linsize4>0))
354  {
355  for (int cnt1=0; cnt1<linsize1; cnt1++)
356  for (int cnt2=0; cnt2<linsize2; cnt2++)
357  for (int cnt3=0; cnt3<linsize3; cnt3++)
358  for (int cnt4=0; cnt4<linsize4; cnt4++)
359  mem1[cnt1 + linsize1 * ( cnt2 + linsize2 * (cnt3 + linsize3 * cnt4) ) ]
360  = ham_rot.getVmat(index.getNstart(irrep1) + cnt1,index.getNstart(irrep2) + cnt2, index.getNstart(irrep3) + cnt3, index.getNstart(irrep4) + cnt4 );
361 
362  int rightdim = linsize2 * linsize3 * linsize4;
363  // (ijkl) -> (ajkl)
364  if(irrep1 == irrep)
365  for(int d=0;d<rightdim;d++)
366  {
367  const double tmpk = mem1[d*linsize1+k2];
368  const double tmpl = mem1[d*linsize1+l2];
369

371  mem1[d*linsize1+l2] = cos*tmpl+sin*tmpk;
372  }
373

375  // (ajkl) -> (ajkd)
376  if(irrep4 == irrep)
377  for(int d=0;d<leftdim;d++)
378  {
379  const double tmpk = mem1[k2*leftdim+d];
380  const double tmpl = mem1[l2*leftdim+d];
381 
382  mem1[k2*leftdim+d] = cos*tmpk-sin*tmpl;
383  mem1[l2*leftdim+d] = cos*tmpl+sin*tmpk;
384  }
385

387  leftdim = linsize1 * linsize2;

391  for(int d=0;d<leftdim;d++)
